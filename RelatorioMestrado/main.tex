% Robust placedotsbetween that computes absolute coords (no polar + relative)
% \placedotsbetween{blob}{vstart}{vend}{radius}{N}
\newcommand{\placedotsbetweeN}[5]{%
  \begin{scope}
    % --- read absolute center coordinates (in pt) ---
    \pgfpointanchor{#1}{center}\relax
    \pgfmathsetmacro{\pbx}{\pgf@x/1pt}\relax
    \pgfmathsetmacro{\pby}{\pgf@y/1pt}\relax

    \pgfpointanchor{#2}{center}\relax
    \pgfmathsetmacro{\pax}{\pgf@x/1pt}\relax
    \pgfmathsetmacro{\pay}{\pgf@y/1pt}\relax

    \pgfpointanchor{#3}{center}\relax
    \pgfmathsetmacro{\pcx}{\pgf@x/1pt}\relax
    \pgfmathsetmacro{\pcy}{\pgf@y/1pt}\relax

    % --- vectors from blob center to start/end (numeric in pt) ---
    \pgfmathsetmacro{\vdxA}{\pax - \pbx}\relax
    \pgfmathsetmacro{\vdyA}{\pay - \pby}\relax
    \pgfmathsetmacro{\vdxC}{\pcx - \pbx}\relax
    \pgfmathsetmacro{\vdyC}{\pcy - \pby}\relax

    % --- angles (degrees) ---
    \pgfmathsetmacro{\angA}{atan2(\vdyA,\vdxA)}\relax
    \pgfmathsetmacro{\angC}{atan2(\vdyC,\vdxC)}\relax

    % --- shortest delta normalized to (-180,180] ---
    \pgfmathsetmacro{\rawdelta}{\angC - \angA}\relax
    \pgfmathsetmacro{\delta}{\rawdelta}\relax
    \ifdim\delta pt>180pt \pgfmathsetmacro{\delta}{\delta - 360}\fi
    \ifdim\delta pt<-180pt \pgfmathsetmacro{\delta}{\delta + 360}\fi

    % --- step angle: put N dots between endpoints (not including endpoints) ---
    \pgfmathsetmacro{\Nd}{#5}\relax
    \pgfmathsetmacro{\step}{\delta/(\Nd+1)}\relax

    % --- radius numeric in pt ---
    \pgfmathsetlength{\Rlen}{#4}\relax
    \pgfmathsetmacro{\Rpt}{\Rlen/1pt}\relax

    % --- place N dots using absolute x,y computed numerically ---
    \foreach \k in {1,...,#5} {%
      \pgfmathsetmacro{\angk}{\angA + \k*\step}\relax
      \pgfmathsetmacro{\xpt}{\pbx + \Rpt * cos(\angk)}\relax
      \pgfmathsetmacro{\ypt}{\pby + \Rpt * sin(\angk)}\relax
      \node[inner sep=0pt, draw=none] at (\xpt pt,\ypt pt) {\(\cdot\)};%
    }%
  \end{scope}%
}
\makeatother

\newcommand{\placedotsbetween}[5]{%
  \begin{scope}
    % blob center
    \pgfpointanchor{#1}{center}%
    \pgfmathsetmacro{\bxpt}{\pgf@x/1pt}%
    \pgfmathsetmacro{\bypt}{\pgf@y/1pt}%

    % start vertex
    \pgfpointanchor{#2}{center}%
    \pgfmathsetmacro{\axpt}{\pgf@x/1pt}%
    \pgfmathsetmacro{\aypt}{\pgf@y/1pt}%

    % end vertex
    \pgfpointanchor{#3}{center}%
    \pgfmathsetmacro{\cxpt}{\pgf@x/1pt}%
    \pgfmathsetmacro{\cypt}{\pgf@y/1pt}%

    % vectors
    \pgfmathsetmacro{\dxA}{\axpt - \bxpt}%
    \pgfmathsetmacro{\dyA}{\aypt - \bypt}%
    \pgfmathsetmacro{\dxC}{\cxpt - \bxpt}%
    \pgfmathsetmacro{\dyC}{\cypt - \bypt}%

    % angles
    \pgfmathsetmacro{\angA}{atan2(\dyA,\dxA)}%
    \pgfmathsetmacro{\angC}{atan2(\dyC,\dxC)}%

    % shortest arc
    \pgfmathsetmacro{\rawdelta}{\angC - \angA}%
    \pgfmathsetmacro{\delta}{\rawdelta}%
    \ifdim\delta pt>180pt \pgfmathsetmacro{\delta}{\delta - 360}\fi
    \ifdim\delta pt<-180pt \pgfmathsetmacro{\delta}{\delta + 360}\fi

    % spacing
    \pgfmathsetmacro{\step}{\delta/(#5+1)}%

    % place dots
    \foreach \k in {1,...,#5} {
      \pgfmathsetmacro{\angk}{\angA + \k*\step}%
      \node[inner sep=0pt, draw=none] 
        at ($ (#1) + (\angk:#4) $) {\(\cdot\)};
    }
  \end{scope}
}
\makeatother

\tikzset{
  cut/.style={
    postaction={
      decorate,
      decoration={
        markings,
        mark= at position 0.5 with {
          % draw a short line perpendicular to the path
          \pgftransformrotate{\pgfdecoratedangle} % align to tangent
          \pgftransformrotate{90}                   % rotate to perpendicular
          \draw[thick, red, dashed, dash pattern=on 3pt off 1pt] (-0.2cm,0) -- (0.2cm,0);
        }
      }
    }
  }
}

\tikzset{
  cut2/.style={
    postaction={
      decorate,
      decoration={
        markings,
        mark= at position 0.5 with {
          % compute angle (normal to path = decoratedangle + 90)
          \pgfmathsetmacro{\ang}{\pgfdecoratedangle + 90}
          % convert half-length and gap to pt lengths for pgf math
          \pgfmathsetmacro{\L}{0.2}   % half-length
          % compute dx,dy for half-length
          \pgfmathsetmacro{\dx}{\L*cos(\ang)}
          \pgfmathsetmacro{\dy}{\L}
          % draw two short segments leaving a central gap
          \draw[thick, red, dashed, dash pattern=on 3pt off 1pt] (0 cm,-\dy cm) -- (0 cm,\dy cm);
        }
      }
    }
  }
}
